/* punto in movimento circolare */
#include <allegro.h> 
#include <math.h> 
int main ()
{

    allegro_init();
    install_keyboard();
    set_color_depth(16);
    int ret = set_gfx_mode(GFX_AUTODETECT_WINDOWED, 800, 600, 0, 0);
    
    if(ret != 0)
    {
        allegro_message("Could not set graphics mode");
        allegro_exit();
    }
    
    
    do  
    {
        x = sin(angle)*radius + x_offset;
        y = cos(angle)*radius + y_offset;
        putpixel(screen, x, y, color);
        angle += 0.00001;
   } while (!key[KEY_ESC]); 
   
    allegro_exit();
  return 0;
}
