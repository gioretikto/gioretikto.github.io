Arrays of structure

E' possibile definire array di strutture

#include <stdio.h>

struct table {
	_Bool id;
	int points;
	int hand;
};

void init_struct (struct table *player);
void print_struct (struct table *player);

int main() {

	struct table player[2];
	
	player[0].id = 0;
	player[1].id = 1;
	
	init_struct(&player[0]);
	init_struct(&player[1]);
	
	print_struct(&player[0]);
	print_struct(&player[1]);
	return 0;
}


void init_struct (struct table *player) {

	if (player->id == 0) {
		player->points = 5;
		player->hand = 30;
	}
	
	else {
		player->points = 10;
		player->hand = 77;
	}
}

void print_struct (struct table *player) {

		if (player->id == 0)
			printf("Dealer's data\n");
		else
			printf("Player's data\n");
			
		printf("hand: %d\n", player->hand);
		printf("hand: %d\n", player->points);
}

#include <stdio.h>

struct student
{
	char *name;
	int age;
};

/* function prototype*/
void display(struct student *s);

int main()
{
	struct student s[2];
	
	s[0].name = "George";
	s[0].age = 36;
	
	s[1].name ="Bond";
	s[1].age = 50;
	
	display(s);

	return 0;
}

void display(struct student *s) 
{
	int i;
	
	for (i=0; i < 2; i++){
	 	  printf("Displaying information\n");
		  printf("Name: %s\n", s[i].name);
		  printf("Age: %d\n", s[i].age);
	}
}
